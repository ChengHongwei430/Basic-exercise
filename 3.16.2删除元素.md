class Solution {
public:
    int removeElement(vector<int>& nums, int val) {

        \\删除指定元素
        int nlen=nums.size();
        if (nlen<=0) return 0 ;

        for(int i=0;i<nlen;++i){
            if(nums[i]==val){
                for(int j=i+1;j<nlen;++j){
                    nums[j-1]=nums[j];
                }

                nums.pop_back();\\删除数组尾部元素

                --i;//每次更新数组后会将后面一个未被检测的移到前面
                --nlen;//数组的长度在缩小，需要及时更新
            }
        }
        return nlen;

    }
};
