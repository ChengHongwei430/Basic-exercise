```c++
class Solution {

public:

  int search(vector<int>& nums, int target) {

​    //二分法复习

​    int nlen=nums.size();

​    if (nlen<=0){

​      return -1;

​    }

​    int low=0;

​    int high=nlen-1;

​    while(low<=high){

​      int mid=(low+high)/2;

​      if (nums[mid]==target) return mid;

​      else if (nums[mid]<target) low=mid+1;

​      else high=mid-1;

​    }

​    return -1;     

  }

};
```
