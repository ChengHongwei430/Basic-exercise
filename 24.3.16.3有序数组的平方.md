```C++
class Solution {
public:
    vector<int> sortedSquares(vector<int>& nums) {

        //有序数组的平方
        int nlen=nums.size();

        for (int i=0;i<nlen;++i) {
            nums[i]=nums[i]*nums[i];
            //不要写nums[i]^2,在c++中代表按比特操作按位异或
        }
        sort(nums.begin(), nums.end());
        //开始是begin,不是begain,again是再一次
        return nums;

    }
};
```
